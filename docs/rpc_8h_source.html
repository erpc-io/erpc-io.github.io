522  inline size_t in_order_client(const SSlot *sslot, const pkthdr_t *pkthdr) {
523  // Counters for pkthdr's request number are valid only if req numbers match
524  if (unlikely(pkthdr->req_num != sslot->cur_req_num)) return false;
525 
526  const auto &ci = sslot->client_info;
527  if (unlikely(pkthdr->pkt_num != ci.num_rx)) return false;
528 
529  // Ignore spurious packets received as a consequence of rollback:
530  // 1. We've only sent pkts up to (ci.num_tx - 1). Ignore later packets.
531  // 2. Ignore if the corresponding client packet for pkthdr is still in wheel
532  if (unlikely(pkthdr->pkt_num >= ci.num_tx)) return false;
533 
534  if (kCcPacing && unlikely(ci.in_wheel[pkthdr->pkt_num % kSessionCredits])) {
535  pkt_loss_stats.still_in_wheel_during_retx++;
536  return false;
537  }
538 
539  return true;
540  }

738  inline void do_tx_burst_st() {
739  assert(in_dispatch());
740  assert(tx_batch_i > 0);
741 
742  // Measure TX burst size
743  dpath_stat_inc(dpath_stats.tx_burst_calls, 1);
744  dpath_stat_inc(dpath_stats.pkts_tx, tx_batch_i);
745 
746  if (kCcRTT) {
747  size_t batch_tsc = 0;
748  if (kCcOptBatchTsc) batch_tsc = dpath_rdtsc();
749 
750  for (size_t i = 0; i < tx_batch_i; i++) {
751  if (tx_burst_arr[i].tx_ts != nullptr) {
752  *tx_burst_arr[i].tx_ts = kCcOptBatchTsc ? batch_tsc : dpath_rdtsc();
753  }
754  }
755  }
756 
757  transport->tx_burst(tx_burst_arr, tx_batch_i);
758  tx_batch_i = 0;
759  }
