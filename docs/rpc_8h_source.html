546  inline size_t in_order_client(const SSlot *sslot, const pkthdr_t *pkthdr) {
547  // Counters for pkthdr's request number are valid only if req numbers match
548  if (unlikely(pkthdr->req_num_ != sslot->cur_req_num_)) return false;
549 
550  const auto &ci = sslot->client_info_;
551  if (unlikely(pkthdr->pkt_num_ != ci.num_rx_)) return false;
552 
553  // Ignore spurious packets received as a consequence of rollback:
554  // 1. We've only sent pkts up to (ci.num_tx - 1). Ignore later packets.
555  // 2. Ignore if the corresponding client packet for pkthdr is still in wheel
556  if (unlikely(pkthdr->pkt_num_ >= ci.num_tx_)) return false;
557 
558  if (kCcPacing &&
559  unlikely(ci.in_wheel_[pkthdr->pkt_num_ % kSessionCredits])) {
560  pkt_loss_stats_.still_in_wheel_during_retx_++;
561  return false;
562  }
563 
564  return true;
565  }

800  }
